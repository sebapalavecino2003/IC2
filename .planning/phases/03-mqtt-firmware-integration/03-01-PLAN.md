    Actually, simplify: the MQTT task publishes every 10s (using vTaskDelay). On each 10s cycle, it checks the most recently known readings and publishes them. The sensor readings arrive from the queue at varying frequencies (DHT22 at 2s, MQ-9 at 1s, KY-026 at 200ms). So the MQTT task accumulates the latest reading per sensor type as they arrive, and at publish time constructs the full payload.

    Implementation approach for the task function:
    - Four local `SensorReading` accumulators: `last_temp`, `last_hum`, `last_gas`, `last_flame`, all initialized as default SensorReading()
    - Main loop:
      ```
      while (1) {
          // Try to receive a sensor reading (100ms timeout)
          SensorReading reading;
          if (xQueueReceive(m_sensor_queue, &reading, pdMS_TO_TICKS(100)) == pdTRUE) {
              // Update accumulator based on reading.type
              switch (reading.type) {
                  case DHT22_TEMPERATURE: last_temp = reading; break;
                  case DHT22_HUMIDITY:    last_hum  = reading; break;
                  case MQ9_GAS:           last_gas  = reading; break;
                  case KY026_FLAME:       last_flame= reading; break;
              }
          }
          
          // Every ~10s, publish telemetry (check using esp_timer)
          uint32_t now_ms = (uint32_t)(esp_timer_get_time() / 1000);
          static uint32_t last_publish_ms = 0;
          if ((now_ms - last_publish_ms) >= 10000) {
              if (m_connected && m_client) {
                  // Build JSON from accumulators and publish
                  char payload[256];
                  snprintf(payload, sizeof(payload),
                      "{\"device_id\":\"%s\",\"timestamp\":%u,\"temperature\":%.1f,"
                      "\"humidity\":%.1f,\"gas_ppm\":%.0f,\"flame_detected\":%u}",
                      DEVICE_ID, now_ms,
                      last_temp.value, last_hum.value, last_gas.value,
                      (last_flame.value > 0.5f) ? 1u : 0u);
                  
                  char topic[64];
                  buildTopic("telemetry", topic, sizeof(topic));
                  
                  int ret = esp_mqtt_client_publish(m_client, topic, payload, 0, 0, 0);
                  if (ret < 0) {
                      // Publish failed — buffer for later
                      m_buffer.enqueue(topic, payload, 0);
                      m_error_handler->reportError("MQTT", "Publish failed — buffered");
                  }
              } else {
                  // Not connected — buffer the message (if we have data)
                  char topic[64];
                  buildTopic("telemetry", topic, sizeof(topic));
                  char payload[256];
                  snprintf(payload, sizeof(payload),
                      "{\"device_id\":\"%s\",\"timestamp\":%u,\"temperature\":%.1f,"
                      "\"humidity\":%.1f,\"gas_ppm\":%.0f,\"flame_detected\":%u}",
                      DEVICE_ID, now_ms,
                      last_temp.value, last_hum.value, last_gas.value,
                      (last_flame.value > 0.5f) ? 1u : 0u);
                  m_buffer.enqueue(topic, payload, 0);
              }
              last_publish_ms = now_ms;
          }
      }
      ```
